const handleToggleNeighborhood = (neighborhood: string) => {
  setFormSelection((prev) => {
    if (prev.neighborhoods.includes(neighborhood)) {
      return {
        ...prev,
        neighborhoods: prev.neighborhoods.filter((n) => n !== neighborhood),
      };
    } else {
      return {
        ...prev,
        neighborhoods: [...prev.neighborhoods, neighborhood],
      };
    }
  });
};
```

// Modificar o useMemo que agrupa as datas (linha ~XXX):
const groupedByCity = useMemo(() => {
  const groups = new Map<string, Map<string | null, AllowedVisitDate[]>>();

  filteredAllowedDates.forEach((date) => {
    const city = date.city;
    const neighborhood = date.neighborhood || null;

    if (!groups.has(city)) {
      groups.set(city, new Map());
    }

    const cityGroup = groups.get(city)!;
    if (!cityGroup.has(neighborhood)) {
      cityGroup.set(neighborhood, []);
    }

    cityGroup.get(neighborhood)!.push(date);
  });

  return groups;
}, [filteredAllowedDates]);
```
